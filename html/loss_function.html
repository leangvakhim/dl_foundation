import torch
import torch.nn as nn

# 1. Setup True Value (10) and Guess (5)
y_true = torch.tensor([10.0])
y_pred = torch.tensor([5.0], requires_grad=True)

# 2. Define Loss Function (Mean Squared Error)
loss_fn = nn.MSELoss()

# 3. Calculate Loss (The Square Area!)
loss = loss_fn(y_pred, y_true)
print(f"Loss: {loss.item()}") # Output: 25.0

# 4. Find the Slope (Gradient)
loss.backward()
print(f"Slope: {y_pred.grad.item()}") # Output: -10.0

# 5. Take a step downhill (Learning!)
lr = 0.1
with torch.no_grad():
    y_pred -= lr * y_pred.grad

print(f"New Guess: {y_pred.item()}") # Closer to 10!
Prediction ($\hat{y}$): 5.0 Loss: 25.0
